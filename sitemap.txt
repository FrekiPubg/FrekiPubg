   name: Generate Sitemap
   on:
     push:
       branches:
         - main
   jobs:
     build:
       runs-on: ubuntu-latest
       steps:
         - name: Checkout code
           uses: actions/checkout@v2
         - name: Generate Sitemap
           uses: cicirello/generate-sitemap@v1
           with:
             base-url-path: "https://frekipubg.github.io/FrekiPubg/"
